FIX 5: Interactive Wikipedia Entities in `components/exhibit-view.tsx`
- Install `react-markdown`.
- Import it into the exhibit view component.
- Instead of rendering the 'Inner World' and 'The Moment' texts as raw strings, render them using the following `ReactMarkdown` configuration so that any bolded text generated by the LLM becomes a seamless Wikipedia Special:Search link. 
- Ensure you pass the track's effective `language` (e.g., 'ru', 'en') from the API payload into this render function so the Wikipedia URL routes to the correct localized subdomain.

```tsx
import ReactMarkdown from 'react-markdown';

// Helper function to use inside the component for rendering the LLM text
const renderMarkdown = (text: string, lang: string) => {
  return (
    <ReactMarkdown
      components={{
        strong: ({ node, ...props }) => {
          const entityText = props.children?.toString() || "";
          const wikiUrl = `https://${lang}.wikipedia.org/wiki/Special:Search?search=${encodeURIComponent(entityText)}`;
          
          return (
            <a 
              href={wikiUrl} 
              target="_blank" 
              rel="noopener noreferrer"
              className="font-bold text-ink dark:text-chalk underline decoration-warm-grey/40 underline-offset-4 transition-colors hover:decoration-ink dark:hover:decoration-chalk"
              title={`Search Wikipedia for ${entityText}`}
            >
              {props.children}
            </a>
          );
        }
      }}
    >
      {text}
    </ReactMarkdown>
  );
};
